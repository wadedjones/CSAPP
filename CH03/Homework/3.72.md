### The following code is for a function similar to `vfunct`

```c
#include <alloca.h>

long aframe(long n, long idx, long *q) {
    long i;
    long **p = alloca(n * sizeof(long *));
    p[0] = &i;
    for (i = 1; i < n; i++) {
        p[i] = q;
    }
    return *p[idx];
}
```

```asm

    leaq    30(,%rdi,8), %rax

    subq    %rax, %rsp          // allocate space for array p (%rsp = s2)
    leaq    15(%rsp), %r8
    andq    $-16, %r8           // set %r8 to &p[0]
```
